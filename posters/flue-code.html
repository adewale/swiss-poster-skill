import { Agent, Model, tool, skill, sandbox }
  from "@flue/runtime";
import { z } from "zod";

// the model reasons over the goal and picks each action
const model = Model("anthropic/claude-sonnet-4-6");

// typed tools — the only way the agent touches the world
const openPR = tool({
  name: "open_pull_request",
  input: z.object({ title: z.string(), body: z.string() }),
  run: async (i) => await gh.pulls(i),
});

// give the agent equipment, expertise, and a safe place to work
const agent = new Agent({
  model,
  tools: [openPR, readFile, runTests],
  skills: [skill("./skills/review/SKILL.md")],
  sandbox: sandbox({ tier: "local", fs: true, shell: true }),
});

// a named session carries state & memory across turns
const session = await agent.session("triage-1842");

// hand it a goal, not a script — then let the loop run
await session.run(`Find the failing test on this branch,
  fix the cause, and open a pull request.`);
